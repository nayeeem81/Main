function showNotification(message, type = 'error', duration = 4000) {
  const container = document.getElementById('notification-container');
  if (!container) return;

  // Create toast element
  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
  
  // Structure internal content
  toast.innerHTML = `
    <span class="toast-message">${message}</span>
    <button class="toast-close" onclick="this.parentElement.remove()">&times;</button>
  `;

  // Append to DOM
  container.appendChild(toast);

  // Trigger CSS entry transition
  setTimeout(() => toast.classList.add('show'), 10);

  // Auto-dismiss logic
  setTimeout(() => {
    toast.classList.remove('show');
    // Wait for transition to finish before removing from DOM
    toast.addEventListener('transitionend', () => toast.remove());
  }, duration);
}

// 2. Network Request Handler & Error Extractor
async function makeRequest(url) {
  try {
    const response = await fetch(url);

    // Check if the HTTP status code is NOT in the 200-299 range
    if (!response.ok) {
      // Client errors (400-499) vs Server errors (500+)
      const notificationType = response.status >= 500 ? 'error' : 'warning';
      showNotification(`Error ${response.status}: ${response.statusText}`, notificationType);
      return;
    }

    // If successful (200-299)
    showNotification(`Success! Status code: ${response.status}`, 'success');
    const data = await response.json().catch(() => ({}));
    return data;

  } catch (error) {
    // Catches network failures, CORS issues, or broken URLs where no response is received
    showNotification(`Network Error: ${error.message}`, 'error');
  }
}
