To identify a subdomain for a multi-tenant application, you need to extract the clean host string using _request.Host.Host and split it by the dot (.) character. Here is the production-ready code and architecture pattern to handle this safely.

Step 1: The Subdomain Extraction Code

Add this method to your custom service. It handles edge cases like local development (localhost), missing domains, and stripping out the main domain.

public string? GetSubdomain(IHttpContextAccessor httpContextAccessor)
{
    // 1. Get the clean host name (e.g., "://sample.com" or "localhost")
    string host = httpContextAccessor.HttpContext?.Request.Host.Host ?? string.Empty;

    if (string.IsNullOrWhiteSpace(host) || host == "localhost")
    {
        return null; 
    }

    // 2. Split the host by dots
    string[] segments = host.Split('.');

    // 3. Check if a subdomain exists 
    // For "://sample.com", segments.Length is 3.
    // For "sample.com", segments.Length is 2.
    if (segments.Length > 2)
    {
        string subdomain = segments[0].ToLowerScope();

        // 4. Ignore "www" if users type www.://sample.com
        if (subdomain == "www" && segments.Length > 3)
        {
            return segments[1].ToLowerScope();
        }

        if (subdomain == "www")
        {
            return null; // It was just ://sample.com
        }

        return subdomain;
    }

    return null; // No subdomain found
}
