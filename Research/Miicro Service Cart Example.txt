

To handle a multi-tenant setup where your MVC app uses cookie authentication, the best practice is to secure your Web API using JSON Web Tokens (JWT).
Since your MVC app handles the login cookies, it will act as a trusted bridge: it handles the user session, generates or passes an encrypted token, or securely speaks to the API on behalf of the user. For a clean separation of microservices, your API should read the Tenant ID directly from the JWT claims to isolate data automatically.
## Architecture Answers

* Buyer Roles: Your mapping is correct. The buyer holds the global Identity role User and the tenant-specific policy role Member.
* Guest vs Registered Carts: It is highly recommended to allow users to add to cart before registering. Requiring registration upfront kills sales conversions. You can generate a temporary GUID string for a guest's UserId in the browser cookie/session, and then "merge" or reassign that cart to their real Identity UserId once they log in or register.

## 1. The Multi-Tenant Entities & Context
Every database record must now be tied to a TenantId.

namespace ShoppingCartApi.Entities;
public interface ITenantEntity
{
    string TenantId { get; set; }
}
public class Cart : ITenantEntity
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public string TenantId { get; set; } = string.Empty; // Scopes cart to tenant
    public string UserId { get; set; } = string.Empty;   // Can be a real UserID or a Guest GUID
    public List<CartItem> Items { get; set; } = new();
    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;
    public DateTime UpdatedAt { get; set; } = DateTime.UtcNow;
}
public class CartItem
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public Guid CartId { get; set; }
    public string ProductId { get; set; } = string.Empty;
    public string ProductName { get; set; } = string.Empty;
    public decimal UnitPrice { get; set; }
    public int Quantity { get; set; }
    public string ImageUrl { get; set; } = string.Empty;
}

## EF Core Multi-Tenant Global Query Filter
We extract the TenantId from the current HTTP request claims context and inject it directly into the Entity Framework Context to automatically filter all queries.

using Microsoft.EntityFrameworkCore;using ShoppingCartApi.Entities;using System.Security.Claims;
namespace ShoppingCartApi.Data;
public class CartDbContext : DbContext
{
    private readonly string _tenantId;

    public CartDbContext(DbContextOptions<CartDbContext> options, IHttpContextAccessor httpContextAccessor) 
        : base(options) 
    {
        // Automatically extract Tenant ID from the JWT token claims passed from MVC app
        _tenantId = httpContextAccessor.HttpContext?.User?.FindFirst("tenant_id")?.Value ?? string.Empty;
    }

    public DbSet<Cart> Carts => Set<Cart>();
    public DbSet<CartItem> CartItems => Set<CartItem>();

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // Enforce Multi-Tenancy: Only fetch items matching the current token's Tenant ID
        modelBuilder.Entity<Cart>().HasQueryFilter(c => c.TenantId == _tenantId);

        // Composite unique index: A user can only have ONE cart per specific tenant
        modelBuilder.Entity<Cart>()
            .HasIndex(c => new { c.TenantId, c.UserId })
            .IsUnique();

        modelBuilder.Entity<Cart>()
            .HasMany(c => c.Items)
            .WithOne()
            .HasForeignKey(i => i.CartId)
            .OnDelete(DeleteBehavior.Cascade);
    }

    public override Task<int> SaveChangesAsync(CancellationToken cancellationToken = default)
    {
        // Automatically assign the Tenant ID to new entries before saving to SQL Server
        foreach (var entry in ChangeTracker.Entries<ITenantEntity>().Where(e => e.State == EntityState.Added))
        {
            entry.Entity.TenantId = _tenantId;
        }
        return base.SaveChangesAsync(cancellationToken);
    }
}

## 2. Multi-Tenant Repository Layer
Because of the global query filter applied in our DbContext, your repository implementation safely prevents database leaks between tenants without modifying standard queries.

using Microsoft.EntityFrameworkCore;using ShoppingCartApi.Data;using ShoppingCartApi.Entities;
namespace ShoppingCartApi.Repositories;
public interface ICartRepository
{
    Task<Cart?> GetByUserIdAsync(string userId);
    Task CreateAsync(Cart cart);
    Task SaveChangesAsync();
}
public class CartRepository : ICartRepository
{
    private readonly CartDbContext _context;

    public CartRepository(CartDbContext context) => _context = context;

    public async Task<Cart?> GetByUserIdAsync(string userId)
    {
        // Automatically filtered by Tenant ID behind the scenes by EF Core!
        return await _context.Carts
            .Include(c => c.Items)
            .FirstOrDefaultAsync(c => c.UserId == userId);
    }

    public async Task CreateAsync(Cart cart)
    {
        await _context.Carts.AddAsync(cart);
    }

    public async Task SaveChangesAsync() => await _context.SaveChangesAsync();
}

## 3. Controller Protected by JWT Roles & Policies
This controller requires a valid JWT token. It enforces that the calling client must hold the User role, and permissions mapping to your tenant Member role.

using Microsoft.AspNetCore.Authorization;using Microsoft.AspNetCore.Mvc;using ShoppingCartApi.DTOs;using ShoppingCartApi.Services;
namespace ShoppingCartApi.Controllers;

[ApiController]
[Route("api/[controller]")]
[Authorize(Roles = "User")] // Must be logged in globally as a User
[Authorize(Policy = "TenantMemberPolicy")] // Must meet the tenant specific policypublic class CartController : ControllerBase
{
    private readonly ICartService _cartService;

    public CartController(ICartService cartService) => _cartService = cartService;

    [HttpGet("{userId}")]
    public async Task<ActionResult<CartDto>> GetCart(string userId)
    {
        var cart = await _cartService.GetCartByUserIdAsync(userId);
        return Ok(cart);
    }

    [HttpPost("{userId}/items")]
    public async Task<ActionResult<CartDto>> AddItem(string userId, [FromBody] AddToCartDto dto)
    {
        var cart = await _cartService.AddItemToCartAsync(userId, dto);
        return Ok(cart);
    }
}

## 4. API Program.cs JWT Configuration
Add the JWT Bearer package to your API project (Microsoft.AspNetCore.Authentication.JwtBearer). Configure it to validate the keys generated by your trusted app ecosystem.

using Microsoft.AspNetCore.Authentication.JwtBearer;using Microsoft.IdentityModel.Tokens;using System.Text;using ShoppingCartApi.Data;using ShoppingCartApi.Repositories;using ShoppingCartApi.Services;using Microsoft.EntityFrameworkCore;
var builder = WebApplication.CreateBuilder(args);

builder.Services.AddControllers();
builder.Services.AddHttpContextAccessor();
// Database
builder.Services.AddDbContext<CartDbContext>(options =>
    options.UseSqlServer(builder.Configuration.GetConnectionString("DefaultConnection")));
// Dependency Injection
builder.Services.AddScoped<ICartRepository, CartRepository>();
builder.Services.AddScoped<ICartService, CartService>();
// JWT Authentication Setupvar jwtSettings = builder.Configuration.GetSection("JwtSettings");var secretKey = Encoding.UTF8.GetBytes(jwtSettings["Secret"]!);

builder.Services.AddAuthentication(options =>
{
    options.DefaultAuthenticateScheme = JwtBearerDefaults.AuthenticationScheme;
    options.DefaultChallengeScheme = JwtBearerDefaults.AuthenticationScheme;
})
.AddJwtBearer(options =>
{
    options.TokenValidationParameters = new TokenValidationParameters
    {
        ValidateIssuer = true,
        ValidateAudience = true,
        ValidateLifetime = true,
        ValidateIssuerSigningKey = true,
        ValidIssuer = jwtSettings["Issuer"],
        ValidAudience = jwtSettings["Audience"],
        IssuerSigningKey = new SymmetricSecurityKey(secretKey)
    };
});
// Enforce Policy-based Tenant Roles on incoming JWT tokens
builder.Services.AddAuthorization(options =>
{
    options.AddPolicy("TenantMemberPolicy", policy => 
        policy.RequireClaim("tenant_role", "Member"));
});
var app = builder.Build();
app.UseAuthentication();
app.UseAuthorization();
app.MapControllers();
app.Run();

## 5. How to bridge this from your existing MVC App
Since your MVC app uses cookies, your MVC Controllers will generate a system-to-system microservice JWT token on the backend whenever making calls to the API.
Add a helper method inside your MVC project's API Client Wrapper to forge the short-lived JWT token containing the current tenant context before shooting requests across:

using System.IdentityModel.Tokens.Jwt;using System.Security.Claims;using Microsoft.IdentityModel.Tokens;using System.Text;
public class CartApiClient
{
    private readonly HttpClient _httpClient;
    private readonly IConfiguration _config;

    public CartApiClient(HttpClient httpClient, IConfiguration config)
    {
        _httpClient = httpClient;
        _config = config;
    }

    private string GenerateApiToken(string tenantId, string globalRole, string tenantRole)
    {
        var claims = new[]
        {
            new Claim("tenant_id", tenantId),
            new Claim(ClaimTypes.Role, globalRole), // e.g., "User"
            new Claim("tenant_role", tenantRole)    // e.g., "Member"
        };

        var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["JwtSettings:Secret"]!));
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        var token = new SecurityTokenDescriptor
        {
            Subject = new ClaimsIdentity(claims),
            Expires = DateTime.UtcNow.AddMinutes(5), // Keep it short-lived
            Issuer = _config["JwtSettings:Issuer"],
            Audience = _config["JwtSettings:Audience"],
            SigningCredentials = creds
        };

        var tokenHandler = new JwtSecurityTokenHandler();
        return tokenHandler.WriteToken(tokenHandler.CreateToken(token));
    }

    public async Task<CartDto?> GetCartAsync(string mvcUserId, string currentTenantId)
    {
        // 1. Generate JWT with proper Multi-tenant claims mapping back to user's permissions
        var token = GenerateApiToken(currentTenantId, "User", "Member");

        // 2. Attach Token to header
        _httpClient.DefaultRequestHeaders.Authorization = new AuthenticationHeaderValue("Bearer", token);

        // 3. Request across securely
        var response = await _httpClient.GetAsync($"api/cart/{mvcUserId}");
        return await response.Content.ReadFromJsonAsync<CartDto>();
    }
}
