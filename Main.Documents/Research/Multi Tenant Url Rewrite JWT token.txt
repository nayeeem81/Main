Here is a complete guide and C# implementation in ASP.NET Core that addresses path-rewriting, secure cookie management, and the lifecycle of JWT generation.

Understanding the JWT and URL Lifecycle

To clear up who sets the token and when:

Who sets the JWT? 
Your Backend API creates and signs the JWT token.

When is it set? 
It is generated immediately after a user successfully logs in (provides a valid username and password) for a specific tenant. It is not created when they just type the URL.

Where is it stored? 
The backend wraps the JWT inside a secure HTTP cookie and sends it back to the browser.

What happens when a tenant types their URL directly?
1. If they are not logged in: The path-rewriting middleware extracts the tenant name, realizes there is no auth cookie, and your app redirects them to the login page (/tenantA/login).

2. If they are already logged in: The browser automatically attaches the existing auth cookie to the request. The middleware then extracts the tenant from the URL, extracts the tenant from the cookie, and double-checks that they match.

Step 2: Implement the Path-Rewriting & Cookie Validation Middleware

This middleware does three critical jobs:
1. Extracts the tenant from the URL path (e.g., /tenantA/home \(\rightarrow \) tenantA).
2. Rewrites the path internally so your routing engine sees it as /home.Compares the URL tenant against the encrypted cookie tenant to prevent URL tampering.csharp


using Microsoft.AspNetCore.Http;
using System.IdentityModel.Tokens.Jwt;
using System.Linq;
using System.Threading.Tasks;

public class TenantResolveMiddleware
{
    private readonly RequestDelegate _next;

    public TenantResolveMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task InvokeAsync(HttpContext context, TenantContext tenantContext)
    {
        var path = context.Request.Path.Value;

        // 1. Extract tenant from path segments (e.g., /tenant1/api/products)
        var segments = path.Split('/', StringSplitOptions.RemoveEmptyEntries);

        if (segments.Length > 0)
        {
            string extractedTenant = segments[0];

            // Ignore static files or health checks if needed
            if (extractedTenant != "api" && !extractedTenant.Contains("."))
            {
                tenantContext.TenantId = extractedTenant;

                // 2. Perform internal Path Rewriting
                // Strips "/tenant1" from the beginning so controllers don't need to know about it
                var remainingPath = "/" + string.Join('/', segments.Skip(1));
                context.Request.Path = remainingPath + context.Request.Path.QueryString;

                // 3. Security Cross-Check with Auth Cookie
                if (context.Request.Cookies.TryGetValue("AuthToken", out string jwtToken))
                {
                    var handler = new JwtSecurityTokenHandler();
                    if (handler.CanReadToken(jwtToken))
                        var token = handler.ReadJwtToken(jwtToken);
                        var cookieTenant = token.Claims.FirstOrDefault(c => c.Type == "TenantId")?.Value;

                        // Double check: Prevent user from changing URL to access another tenant
                        if (!string.IsNullOrEmpty(cookieTenant) && 
                            !string.Equals(cookieTenant, extractedTenant, StringComparison.OrdinalIgnoreCase))
                        {
                            context.Response.StatusCode = StatusCodes.Status403Forbidden;
                            await context.Response.WriteAsync("Tenant mismatch tracking detected.");
                            return;
                        }
                    }
                }
            }
        }

        await _next(context);
    }
}

Step 3: Login Controller (Where the JWT is born and set)

This is the API controller handling authentication. It verifies credentials, builds the JWT containing the TenantId claim, and embeds it into a Secure, HttpOnly Cookie.csharp

using Microsoft.AspNetCore.Mvc;
using Microsoft.IdentityModel.Tokens;
using System;
using System.IdentityModel.Tokens.Jwt;
using System.Security.Claims;
using System.Text;

[ApiController]
[Route("api/auth")] // Becomes /tenant1/api/auth before rewrite
public class AuthController : ControllerBase
{
    private readonly TenantContext _tenantContext;

    public AuthController(TenantContext tenantContext)
    {
        _tenantContext = tenantContext;
    }

    [HttpPost("login")]
    public IActionResult Login([FromBody] LoginModel model)
    {
        // 1. Validate your user credentials against the database here...
        // Ensure user belongs to the current _tenantContext.TenantId
        
        // 2. Create claims, explicitly baking the TenantId into the identity
        var claims = new[]
        {
            new Claim(ClaimTypes.Name, model.Username),
            new Claim("TenantId", _tenantContext.TenantId), // Crucial for multi-tenancy
            new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString())
        };

        var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes("YourSuperSecretBackingKeyMustBeLongEnough"));
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        var token = new JwtSecurityToken(
            issuer: "YourAppBackend",
            audience: "YourAppClients",
            claims: claims,
            expires: DateTime.UtcNow.AddHours(8),
            signingCredentials: creds
        );

        var tokenString = new JwtSecurityTokenHandler().WriteToken(token);

        // 3. Bake the token into an encrypted HttpOnly cookie sent directly to the browser
        var cookieOptions = new CookieOptions
        {
            HttpOnly = true,   // Prevents JavaScript cross-site scripting (XSS) extraction
            Secure = true,     // Forces cookie to transmit only over HTTPS
            SameSite = SameSiteMode.Strict, // Protects against CSRF attacks
            Expires = DateTime.UtcNow.AddHours(8),
            Path = $"/{_tenantContext.TenantId}" // Restricts cookie visibility to this tenant's path scope
        };

        Response.Cookies.Append("AuthToken", tokenString, cookieOptions);

        return Ok(new { Message = "Login successful" });
    }
}

public class LoginModel 
{ 
    public string Username { get; set; } 
    public string Password { get; set; } 
}

Step 4: Injecting Tenant & JWT Claims Into Your Logs

To view structural logs tracking precisely which user and tenant performed an action, append them to the logging scope during the Authentication configuration in Program.cs.csharp

using Microsoft.AspNetCore.Authentication.JwtBearer;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.IdentityModel.Tokens;
using Serilog; // Recommended logging library for structural properties
using System.Text;

var builder = WebApplication.CreateBuilder(args);

// Register Tenant Data Containers
builder.Services.AddScoped<TenantContext>();

builder.Services.AddAuthentication(options =>
{
    options.DefaultAuthenticateScheme = JwtBearerDefaults.AuthenticationScheme;
    options.DefaultChallengeScheme = JwtBearerDefaults.AuthenticationScheme;
})
.AddJwtBearer(options =>
{
    // Instruct app to read the token out of your secure cookie instead of standard Headers
    options.Events = new JwtBearerEvents
    {
        OnMessageReceived = context =>
        {
            if (context.Request.Cookies.TryGetValue("AuthToken", out string token))
            {
                context.Token = token;
            }
            return Task.CompletedTask;
        },
        OnTokenValidated = context =>
        {
            // PUSH CLAIMS TO LOGGING SCOPE FOR TRACKING
            var logger = context.HttpContext.RequestServices.GetRequiredService<ILogger<Program>>();
            
            var tenantId = context.Principal?.FindFirst("TenantId")?.Value;
            var username = context.Principal?.Identity?.Name;

            // Creates a structured logging boundary carrying contextual identity properties
            Serilog.Context.LogContext.PushProperty("Log_TenantId", tenantId);
            Serilog.Context.LogContext.PushProperty("Log_Username", username);

            return Task.CompletedTask;
        }
    };

    options.TokenValidationParameters = new TokenValidationParameters
    {
        ValidateIssuer = true,
        ValidateAudience = true,
        ValidateLifetime = true,
        ValidateIssuerSigningKey = true,
        ValidIssuer = "YourAppBackend",
        ValidAudience = "YourAppClients",
        IssuerSigningKey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes("YourSuperSecretBackingKeyMustBeLongEnough"))
    };
});

builder.Services.AddControllers();

var app = builder.Build();

// CRITICAL: Middleware Order sequence matters!

app.UseMiddleware<TenantResolveMiddleware>(); 
// 1. Run rewrite first to determine tenant layout

app.UseAuthentication(); 
// 2. Authenticate the token found inside the cookie

app.UseAuthorization();  
// 3. Enforce authorization barriers

app.MapControllers();

app.Run();
