builder.Services.AddCors(options =>
{
    options.AddDefaultPolicy(policy =>
    {
        policy.AllowAnyHeader()
              .AllowAnyMethod()
              .AllowCredentials()
              .SetIsOriginAllowed((origin) =>
              {
                  // A. GLOBAL FALLBACK: Always allow your internal admin tools
                  var globalAllowedOrigins = new[] { "https://internalcompany.com" };
                  if (globalAllowedOrigins.Contains(origin)) return true;

                  // B. TENANT SPECIFIC: Fallback to checking the active tenant if it's not the admin tool
                  var httpContextAccessor = builder.Services.BuildServiceProvider()
                                                   .GetRequiredService<IHttpContextAccessor>();
                  var httpContext = httpContextAccessor.HttpContext;
                  if (httpContext == null) return false;

                  // Simply pull what the middleware already found
                  var tenantProvider = httpContext.RequestServices.GetRequiredService<ITenantProvider>();
                  string? tenantId = tenantProvider.TenantId;
                  if (string.IsNullOrEmpty(tenantId)) return false;

                  var configuration = httpContext.RequestServices.GetRequiredService<IConfiguration>();
                  var allowedOrigins = configuration
                      .GetSection($"TenantCorsSettings:{tenantId}")
                      .Get<List<string>>();

                  return allowedOrigins != null && allowedOrigins.Contains(origin);
              });
    });
});
