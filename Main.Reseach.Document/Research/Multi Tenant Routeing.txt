Setting up a route constraint for /{tenant}/[controller] ensures that ASP.NET Core only matches the route if the {tenant} segment exists in your database or configuration. 


This prevents standard static paths (like /images or /api) from accidentally being treated as tenant names.


Here is the step-by-step implementation guide using a custom route constraint.


Step 1: Create the Custom Route Constraint


Create a class that implements IRouteConstraint. This class checks if the requested tenant segment actually exists in your system.csharp


using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Routing;

public class TenantRouteConstraint : IRouteConstraint
{
    // In real apps, inject a service or cache to validate against a database

    private readonly HashSet<string> _validTenants = new(StringComparer.OrdinalIgnoreCase)
    {
        "tenant1-local",
        "tenant2-local",
        "alpha-corp"
    };

    public bool Match(
        HttpContext? httpContext, 
        IRouter? route, 
        string routeKey, 
        RouteValueDictionary values, 
        RouteDirection routeDirection)
    {
        // 1. Ensure the key we are checking is the 'tenant' key

        if (!values.TryGetValue(routeKey, out var value) || value is not string tenantSegment)
        {
            return false;
        }

        // 2. Validate the tenant segment against your allowed list/database

        return _validTenants.Contains(tenantSegment);
    }
}

Use code with caution.

Step 2: Register the Constraint and Set Up Routing

Open your Program.cs file. You need to register your custom constraint with a short nickname (e.g., tenant) and then apply it to your controller routing map.

csharp


var builder = WebApplication.CreateBuilder(args);

builder.Services.AddControllersWithViews(); // or AddControllers();

// Register the custom constraint inline with the Routing options

builder.Services.Configure<RouteOptions>(options =>
{
    options.ConstraintMap.Add("tenant", typeof(TenantRouteConstraint));

});

var app = builder.Build();

app.UseHttpsRedirection();
app.UseStaticFiles();
app.UseRouting();
app.UseAuthorization();

// Map the multi-tenant route pattern using the ':tenant' constraint nickname

app.MapControllerRoute(
    name: "multitenant",
    pattern: "{tenant:tenant}/{controller=Home}/{action=Index}/{id?}");


// Default fallback route for landing pages (e.g., ://khai.com)

app.MapControllerRoute(
    name: "default",
    pattern: "{controller=Home}/{action=Index}/{id?}");

app.Run();

Use code with caution.

To avoid repeating string tenant as a parameter in every single action method, you can create a custom BaseController that intercepts the routing data automatically.

By overriding the OnActionExecuting lifecycle method, the base controller can extract the tenant segment and store it in a globally accessible property or ViewData field for all inheriting controllers.

Step 1: Create the Base Controller

Create an abstract class named TenantBaseController that inherits from Controller.csharp

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Filters;

public abstract class TenantBaseController : Controller
{
    // Accessible by any controller inheriting from this base class
    public string? CurrentTenant { get; private set; }

    public override void OnActionExecuting(ActionExecutingContext context)
    {
        // Extract the {tenant} value calculated by the route mapping
        if (context.RouteData.Values.TryGetValue("tenant", out var tenantObj) && tenantObj is string tenant)
        {
            CurrentTenant = tenant;
            
            // Automatically make it available to all Razor Views layout pages
            ViewData["TenantName"] = tenant;
        }
        else
        {
            // Optional: Redirect to a global landing page if the tenant is missing
            context.Result = new RedirectToActionResult("Index", "Home", new { area = "" });
        }

        base.OnActionExecuting(context);
    }
}

Use code with caution.

To handle multi-tenancy seamlessly across an API layer, you can apply a similar architecture using ControllerBase.

When injecting services into a base controller, using standard constructor injection forces every child controller to declare and pass that dependency up via base(...). To avoid this repetitive boilerplate code, you can use Property Injection via HttpContext inside an Action Filter or an override method. 

This keeps your child controller constructors entirely clean.

Here is the complete implementation for an API Base Controller using property-based dependency injection.

Step 1: Create the API Tenant Base Controller

This base class inherits from ControllerBase and targets the [ApiController] pipeline. It automatically resolves your scoped ITenantContext (which we created in the earlier step) directly out of the request's dependency container.

csharp

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Filters;
using Microsoft.Extensions.DependencyInjection;

[ApiController]
[Route("{tenant}/api/[controller]")] // Global route prefix for all APIs
public abstract class ApiTenantBaseController : ControllerBase, IActionFilter
{
    // Child controllers can access this property directly
    protected string? CurrentTenant { get; private set; }

    [NonAction]
    public void OnActionExecuting(ActionExecutingContext context)
    {
        // 1. Extract tenant string from route
        if (context.RouteData.Values.TryGetValue("tenant", out var tenantObj) && tenantObj is string tenant)
        {
            CurrentTenant = tenant;

            // 2. Inject Scoped Service cleanly without constructor boilerplate
            // This resolves the unique Scoped service instance allocated to this specific HTTP request

            var tenantContext = context.HttpContext.RequestServices.GetRequiredService<ITenantContext>();
            
            // Hydrate the service state

            tenantContext.TenantId = tenant;
        }
        else
        {
            // Short-circuit the request if the tenant route token is missing

            context.Result = new BadRequestObjectResult(new { error = "Tenant context is missing from the request path." });
        }
    }

    [NonAction]
    public void OnActionExecuted(ActionExecutedContext context)
    {
        // No post-execution logic required, but interface implementation is mandatory
    }
}

Use code with caution.

To pass the resolved tenant down to your data access layer (like EF Core) safely, you should inject it into a Scoped Service.In ASP.NET Core, a 

Scoped service is created fresh for every single HTTP request. This makes it the cleanest and safest way to store the tenant context without risking data leaks between different users. While AsyncLocal<T> works well for background threads, a Scoped service natively leverages ASP.NET Core's architecture for dependency injection (DI).

Here is how to set up an end-to-end data isolation pipeline.

Step 1: Create a Tenant Context Container

First, create a scoped service interface and implementation that holds the current tenant's identifier.csharp

public interface ITenantContext
{
    string? TenantId { get; set; }
}

public class TenantContext : ITenantContext
{
    public string? TenantId { get; set; }
}

Use code with caution.

Step 2: Extract the Tenant via Custom Middleware

Create a middleware that runs right after routing executes. At this stage, ASP.NET Core has already validated the route using your TenantRouteConstraint and stored the value in the RouteData.csharp

using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Routing;
using System.Threading.Tasks;

public class TenantResolverMiddleware
{
    private readonly RequestDelegate _next;

    public TenantResolverMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task InvokeAsync(HttpContext context, ITenantContext tenantContext)
    {
        // Get the tenant string extracted from the /{tenant}/ route
        var tenantRouteValue = context.GetRouteValue("tenant") as string;

        if (!string.IsNullOrEmpty(tenantRouteValue))
        {
            // Inject the string into the scoped container for this request
            tenantContext.TenantId = tenantRouteValue;
        }

        // Continue down the execution pipeline
        await _next(context);
    }
}
Use code with caution.

Step 3: Register everything in Program.cs

Register your ITenantContext as Scoped and plug your middleware into the application pipeline. It must be placed after app.UseRouting() so that context.GetRouteValue() can find the tenant segment.csharp

var builder = WebApplication.CreateBuilder(args);

builder.Services.AddControllers();

// Register the TenantContext container as Scoped
builder.Services.AddScoped<ITenantContext, TenantContext>();

var app = builder.Build();

app.UseHttpsRedirection();

app.UseRouting(); // 1. Routing analyzes the URL patterns

app.UseMiddleware<TenantResolverMiddleware>(); 

// 2. Middleware grabs the tenant string

app.UseAuthorization();

app.MapControllers();
app.Run();

Step 4: Enforce Data Isolation in EF Core DbContext

Now you can inject ITenantContext directly into your Entity Framework Core DbContext. By overriding OnModelCreating, you can apply a global query filter that automatically filters out data belonging to other tenants.csharp

using Microsoft.EntityFrameworkCore;

public class ApplicationDbContext : DbContext
{
    private readonly ITenantContext _tenantContext;

    public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options, ITenantContext tenantContext)
        : base(options)
    {
        _tenantContext = tenantContext;
    }

    public DbSet<Product> Products { get; set; } = null!;

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // Configure Query Filter so queries automatically attach: WHERE TenantId = currentTenant
        modelBuilder.Entity<Product>()
            .HasQueryFilter(p => p.TenantId == _tenantContext.TenantId);
    }

    // Automatically stamp the incoming Tenant ID on new records during database saves

    public override Task<int> SaveChangesAsync(CancellationToken cancellationToken = default)
    {
        foreach (var entry in ChangeTracker.Entries<ITenantEntity>())
        {
            if (entry.State == EntityState.Added)
            {
                entry.Entity.TenantId = _tenantContext.TenantId 
                    ?? throw new InvalidOperationException("Tenant identity missing during save.");
            }
        }

        return base.SaveChangesAsync(cancellationToken);
    }
}

// Marker interface for database tables that need tenant isolation
public interface ITenantEntity
{
    string TenantId { get; set; }
}

public class Product : ITenantEntity
{
    public int Id { get; set; }
    public string Name { get; set; } = string.Empty;
    public string TenantId { get; set; } = string.Empty; // DB column
}

Use code with caution.

Why this approach is highly secure

By binding the tenant in middleware and filtering via EF Core, your controller code remains pure. When a developer writes _context.Products.ToList(), EF Core translates that under the hood into SELECT * FROM Products WHERE TenantId = 'tenant1-local'.
