Yes. If you switch to a custom provider or standard cryptography (like HMAC), you can easily combine two variables to generate a token that works both ways (generation and verification). [1, 2] 

Because the default DataProtectorTokenProvider forces a timestamp into the token, you cannot use it to regenerate a perfectly identical string down the road.

## Recommended Architecture: 

HMAC-SHA256 Provider

An HMAC (Hash-based Message Authentication Code) provider perfectly satisfies your requirement. It cryptographically binds your variables together and creates a deterministic token. If you pass the exact same input later, it returns the exact same string. '

[3, 4, 5] 

using System;
using System.Security.Cryptography;
using System.Text;

public class DeterministicTokenEngine
{
    // Keep this safe in your configuration appsettings.json

    private readonly byte[] _secretServerKey;

    public DeterministicTokenEngine(string secretKey)
    {
        _secretServerKey = Encoding.UTF8.GetBytes(secretKey);
    }

    /// <summary>
    /// Generates a unique, reproducible token from a key and tenantId combination.
    /// </summary>

    public string GenerateToken(string key, string tenantId)
    {
        // 1. Combine the variables deterministically 
        // Using a clear delimiter prevents value-bleeding structural bugs

        string combinedInput = $"{tenantId.ToLowerInvariant()}:{key.ToLowerInvariant()}";

        byte[] inputBytes = Encoding.UTF8.GetBytes(combinedInput);

        // 2. Hash using your secure server secret

        using var hmac = new HMACSHA256(_secretServerKey);

        byte[] hashBytes = hmac.ComputeHash(inputBytes);

        // 3. Return a clean, URL-safe string

        return Convert.ToHexString(hashBytes).ToLowerInvariant();
    }

    /// <summary>
    /// Validates the token by regenerating it and checking for a exact match.
    /// </summary>

    public bool ValidateToken(string key, string tenantId, string suppliedToken)
    {
        string expectedToken = GenerateToken(key, tenantId);
        
        // Use a time-constant comparison to protect against timing attacks

        return CryptographicOperations.FixedTimeEquals(
            Encoding.UTF8.GetBytes(expectedToken), 
            Encoding.UTF8.GetBytes(suppliedToken)
        );
    }
}

## Essential Security Precautions

If you build your provider this way, keep these two limitations in mind:

* No Natural Expiration: 

Because this process is completely stateless and lacks a timestamp, tokens never expire. If you need an expiration window, you must alter the combination string to include a time bucket (for example, combining tenantId + key + DateTime.UtcNow.ToString("yyyy-MM-dd") creates a token that naturally cycles and expires every 24 hours). 
[6, 7, 8] 

* Information Exposure Risk: Anyone holding a valid token can run offline brute-force attacks to guess your key or tenantId if your master secretKey is weak. Keep your master server secret long and randomly generated.

[9, 10, 11]

To add an exact 30-minute automatic expiration window while keeping the system stateless, you can append a dynamic time-bucket directly into your combined string.

Because standard clock times switch at different intervals for different users, the industry-standard way to handle this is to divide the current Unix epoch time into 30-minute intervals (blocks).
[1] 

## Time-Window Token Provider

This updated engine checks both the current 30-minute block and the previous 30-minute block. This dual-check prevents valid tokens from suddenly breaking if a user generates a token at 10:29 and submits it at 10:31.

using System;
using System.Security.Cryptography;
using System.Text;

public class ExpiringTokenEngine
{
    private readonly byte[] _secretServerKey;

    // 30 minutes expressed in total seconds

    private const long WindowSeconds = 30 * 60; 

    public ExpiringTokenEngine(string secretKey)
    {
        _secretServerKey = Encoding.UTF8.GetBytes(secretKey);
    }

    /// <summary>
    /// Generates a token valid for the current 30-minute window.
    /// </summary>
    public string GenerateToken(string key, string tenantId)
    {
        long currentWindow = GetCurrentWindowIndex();
        return CreateHash(key, tenantId, currentWindow);
    }

    /// <summary>
    /// Validates the token against the current and previous 30-minute windows.
    /// </summary>
    public bool ValidateToken(string key, string tenantId, string suppliedToken)
    {
        long currentWindow = GetCurrentWindowIndex();

        // Check 1: Is it from the current 30-minute block?

        string currentExpected = CreateHash(key, tenantId, currentWindow);

        if (CryptographicOperations.FixedTimeEquals(Encoding.UTF8.GetBytes(currentExpected), Encoding.UTF8.GetBytes(suppliedToken)))
        {
            return true;
        }

        // Check 2: Is it from the immediately preceding 30-minute block?
        string previousExpected = CreateHash(key, tenantId, currentWindow - 1);

        if (CryptographicOperations.FixedTimeEquals(Encoding.UTF8.GetBytes(previousExpected), Encoding.UTF8.GetBytes(suppliedToken)))
        {
            return true;
        }

        return false;
    }

    private string CreateHash(string key, string tenantId, long windowIndex)
    {
        // Incorporating the windowIndex means the token string changes every 30 minutes

        string combinedInput = $"{tenantId.ToLowerInvariant()}:{key.ToLowerInvariant()}:{windowIndex}";

        byte[] inputBytes = Encoding.UTF8.GetBytes(combinedInput);

        using var hmac = new HMACSHA256(_secretServerKey);
        byte[] hashBytes = hmac.ComputeHash(inputBytes);

        return Convert.ToHexString(hashBytes).ToLowerInvariant();
    }

    private long GetCurrentWindowIndex()
    {
        // Returns how many 30-minute blocks have passed since 1970

        long unixTimeSeconds = DateTimeOffset.UtcNow.ToUnixTimeSeconds();

        return unixTimeSeconds / WindowSeconds;
    }
}

## Critical Rules for 30-Minute Windows

* Actual Lifespan Variance: Because tokens map to strict clock intervals rather than an independent timer, a token will live anywhere from 30 to 60 minutes before expiring. [2] 

* Server Time Sync: Your host servers must have synchronized clocks (using NTP). If Server A's clock drifts by more than a couple of minutes from Server B, tokens generated on one will fail validation on the other.

The best practice for an API is to pass the token via an HTTP Header (specifically the Authorization header), and then unpack it during the request lifecycle to populate the HttpContext.User as Claims. ## 1. The Custom Authentication Middleware'

This middleware intercepts incoming HTTP requests, looks for the token in the headers, validates it, and assigns an identity to the HttpContext. 
[5, 6, 7, 8] 

using Microsoft.AspNetCore.Http;using System.Security.Claims;using System.Threading.Tasks;
public class TenantTokenAuthMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ExpiringTokenEngine _tokenEngine;

    public TenantTokenAuthMiddleware(RequestDelegate next, ExpiringTokenEngine tokenEngine)
    {
        _next = next;
        _tokenEngine = tokenEngine;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        // 1. Extract values from Headers
        // Expecting: Authorization: Bearer <token>
        if (!context.Request.Headers.TryGetValue("Authorization", out var authHeader) || 
            string.IsNullOrEmpty(authHeader))
        {
            await _next(context);
            return;
        }

        string headerValue = authHeader.ToString();
        if (!headerValue.StartsWith("Bearer ", StringComparison.OrdinalIgnoreCase))
        {
            await _next(context);
            return;
        }

        string token = headerValue.Substring("Bearer ".Length).Trim();

        // Expecting the client to also provide context headers to verify against the hash
        context.Request.Headers.TryGetValue("X-Tenant-Id", out var tenantId);
        context.Request.Headers.TryGetValue("X-User-Key", out var userKey);

        if (!string.IsNullOrEmpty(tenantId) && !string.IsNullOrEmpty(userKey))
        {
            // 2. Validate using your 30-minute stateless token engine
            bool isValid = _tokenEngine.ValidateToken(userKey!, tenantId!, token);

            if (isValid)
            {
                // 3. Transform variables into Claims and attach to HttpContext
                var claims = new[]
                {
                    new Claim(ClaimTypes.NameIdentifier, userKey!),
                    new Claim("TenantId", tenantId!),
                    new Claim("AuthToken", token) // Stores the raw token in HttpContext claims if needed later
                };

                var identity = new ClaimsIdentity(claims, "CustomTenantTokenAuth");
                context.User = new ClaimsPrincipal(identity);
            }
        }

        await _next(context);
    }
}

## 3. How to Consume Claims in a Controller
Once the middleware attaches the ClaimsPrincipal to the HttpContext, you can instantly read the keys and tenant details inside any controller action.

using Microsoft.AspNetCore.Authorization;using Microsoft.AspNetCore.Mvc;using System.Linq;

[ApiController]
[Route("api/[controller]")]public class DataController : ControllerBase
{
    [HttpGet("secure-resource")]
    public IActionResult GetSecureData()
    {
        // Check if the middleware authenticated the user
        if (User.Identity == null || !User.Identity.IsAuthenticated)
        {
            return Unauthorized("Missing or invalid tenant token.");
        }

        // Pull the variables directly out of the HttpContext Claims
        string? userKey = User.FindFirst(System.Security.Claims.ClaimTypes.NameIdentifier)?.Value;
        string? tenantId = User.FindFirst("TenantId")?.Value;
        string? tokenUsed = User.FindFirst("AuthToken")?.Value;

        return Ok(new {
            Message = "Access granted via HttpContext Claims!",
            CurrentKey = userKey,
            CurrentTenant = tenantId,
            TokenInstance = tokenUsed
        });
    }
}

## 2. Update the Expiring Token Engine

We modify the token engine so that it no longer stores a single server secret key. Instead, the secret key must be passed dynamically into the GenerateToken and ValidateToken methods at runtime.

using System;
using System.Security.Cryptography;
using System.Text;

public class TenantExpiringTokenEngine
{
    private const long WindowSeconds = 30 * 60; // 30 minutes

    public string GenerateToken(string key, string tenantId, string tenantSecretKey)
    {
        long currentWindow = GetCurrentWindowIndex();

        return CreateHash(key, tenantId, currentWindow, tenantSecretKey);
    }

    public bool ValidateToken
    (string key, string tenantId, string suppliedToken, string tenantSecretKey)
    {
        long currentWindow = GetCurrentWindowIndex();

        // 1. Check current 30-minute block

        string currentExpected = CreateHash(key, tenantId, currentWindow, tenantSecretKey);

        if (CryptographicOperations.FixedTimeEquals
        (Encoding.UTF8.GetBytes(currentExpected), Encoding.UTF8.GetBytes(suppliedToken)))
        {
            return true;
        }

        // 2. Check previous 30-minute block (grace period)

        string previousExpected = CreateHash(key, tenantId, currentWindow - 1, tenantSecretKey);

        if (CryptographicOperations.FixedTimeEquals
        (Encoding.UTF8.GetBytes(previousExpected), Encoding.UTF8.GetBytes(suppliedToken)))
        {
            return true;
        }

        return false;
    }

    private string CreateHash (string key, string tenantId, 
    long windowIndex, string tenantSecretKey)
    {
        string combinedInput = $"{tenantId.ToLowerInvariant()}:{key.ToLowerInvariant()}:{windowIndex}";

        byte[] inputBytes = Encoding.UTF8.GetBytes(combinedInput);

        byte[] secretBytes = Encoding.UTF8.GetBytes(tenantSecretKey);

        using var hmac = new HMACSHA256(secretBytes);

        byte[] hashBytes = hmac.ComputeHash(inputBytes);

        return Convert.ToHexString(hashBytes).ToLowerInvariant();
    }

    private long GetCurrentWindowIndex()
    {
        return DateTimeOffset.UtcNow.ToUnixTimeSeconds() / WindowSeconds;
    }
}

## 3. Update the Middleware to Resolve the Tenant

The middleware now extracts the X-Tenant-Id header, asks the ITenantResolver for that specific tenant's secret key, and passes it to the engine. If the tenant doesn't exist, it short-circuits with a 401 Unauthorized response. 

[1] 

using Microsoft.AspNetCore.Http;
using System;
using System.Security.Claims;
using System.Threading.Tasks;
using Microsoft.Extensions.DependencyInjection;

public class TenantResolverAuthMiddleware
{
    private readonly RequestDelegate _next;
    private readonly TenantExpiringTokenEngine _tokenEngine;

    public TenantResolverAuthMiddleware
    (RequestDelegate next, TenantExpiringTokenEngine tokenEngine)
    {
        _next = next;
        _tokenEngine = tokenEngine;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        // 1. Extract context identification fields from headers

        context.Request.Headers.TryGetValue("X-Tenant-Id", out var tenantIdHeader);

        context.Request.Headers.TryGetValue("X-User-Key", out var userKeyHeader);

        if (string.IsNullOrEmpty(tenantIdHeader) || string.IsNullOrEmpty(userKeyHeader))
        {
            context.Response.StatusCode = StatusCodes.Status401Unauthorized;

            await context.Response.WriteAsync("Missing tracking headers (X-Tenant-Id or X-User-Key).");

            return;
        }

        string tenantId = tenantIdHeader.ToString();

        string userKey = userKeyHeader.ToString();

        // 2. Extract Authorization Bearer Token

        if (!context.Request.Headers.TryGetValue("Authorization", out var authHeader) 
        || string.IsNullOrEmpty(authHeader))
        {
            context.Response.StatusCode = StatusCodes.Status401Unauthorized;

            await context.Response.WriteAsync("Authorization header missing.");

            return;
        }

        string headerValue = authHeader.ToString();

        if (!headerValue.StartsWith("Bearer ", StringComparison.OrdinalIgnoreCase))
        {
            context.Response.StatusCode = StatusCodes.Status401Unauthorized;

            await context.Response.WriteAsync("Authorization scheme must be Bearer.");

            return;
        }

        string token = headerValue.Substring("Bearer ".Length).Trim();

        // 3. Resolve Tenant Secret Key using Scoped Service Provider

        // (Resolvers linked to DbContexts must be resolved within the scope of the request)

        var tenantResolver = context.RequestServices.GetRequiredService<ITenantResolver>();

        string? tenantSecretKey = await tenantResolver.GetTenantSecretKeyAsync(tenantId);

        if (string.IsNullOrEmpty(tenantSecretKey))
        {
            context.Response.StatusCode = StatusCodes.Status401Unauthorized;

            await context.Response.WriteAsync("Invalid or unregistered Tenant ID.");

            return;
        }

        // 4. Validate Token with resolved key

        bool isValid = _tokenEngine.ValidateToken(userKey, tenantId, token, tenantSecretKey);

        if (!isValid)
        {
            context.Response.StatusCode = StatusCodes.Status401Unauthorized;

            await context.Response.WriteAsync("Token validation failed or token expired.");'

            return;
        }

        // 5. Build Claims principal and append to HttpContext

        var claims = new[]
        {
            new Claim(ClaimTypes.NameIdentifier, userKey),

            new Claim("TenantId", tenantId)
        };

        var identity = new ClaimsIdentity(claims, "TenantResolverAuthentication");

        context.User = new ClaimsPrincipal(identity);

        await _next(context);
    }
}
