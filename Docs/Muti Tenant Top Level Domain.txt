### SSL Certificates 
Use a dynamic reverse proxy setup like YARP (Yet Another Reverse Proxy) or Caddy Server to automatically generate and renew Let's Encrypt SSL certificates on the fly for your customers' domains when they hit your site.

To handle automated SSL certificates for an unlimited number of custom customer domains, you cannot use standard static SSL certificates. Instead, your infrastructure must automatically request, provision, and renew Let's Encrypt TLS certificates on the fly whenever a new customer domain routes traffic to your application. The industry-standard approach is to deploy a reverse proxy in front of your ASP.NET Core application to act as the SSL Termination Layer.



# Match any incoming HTTP/HTTPS host header
:443, :80 {
    
    # Enable On-Demand TLS (On-the-fly Let's Encrypt certificates)
    tls {
        on_demand
    }

    # Reverse proxy the clean HTTP traffic to your ASP.NET Core application port
    reverse_proxy localhost:5197
}

using Microsoft.EntityFrameworkCore;

public class TenantResolverMiddleware
{
    private readonly RequestDelegate _next;

    public TenantResolverMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task InvokeAsync(HttpContext context, ITenantSetter tenantSetter, ApplicationDbContext dbContext)
    {
        // 1. Get the current incoming host domain (e.g., "tenant-a.com" or "localhost:5001")
        string hostDomain = context.Request.Host.Host.ToLower();

        // 2. Query the tenant lookup table while bypassing tenant filters
        var tenant = await dbContext.Tenants
            .IgnoreQueryFilters() // Critical! Filters must be off to search all records
            .FirstOrDefaultAsync(t => t.Domain == hostDomain);

        if (tenant != null)
        {
            tenantSetter.CurrentTenantId = tenant.Id;
        }
        else
        {
            // Fallback default domain configuration or redirect to a signup/holding page
            tenantSetter.CurrentTenantId = "default-root"; 
        }

        await _next(context);
    }
}

[ApiController]
[Route("api/domains")]
public class DomainVerificationController : ControllerBase
{
    private readonly ApplicationDbContext _context;

    public DomainVerificationController(ApplicationDbContext context)
    {
        _context = context;
    }

    [HttpGet("check")]
    public async Task<IActionResult> CheckDomain([FromQuery] string domain)
    {
        if (string.IsNullOrWhiteSpace(domain)) return BadRequest();

        // Query database globally, ignoring tenant filters
        bool domainExists = await _context.Tenants
            .IgnoreQueryFilters()
            .AnyAsync(t => t.Domain == domain.ToLower().Trim());

        if (domainExists)
        {
            return Ok(); // Returns HTTP 200: Caddy will issue the SSL certificate
        }

        return NotFound(); // Returns HTTP 404: Caddy will refuse the connection
    }
}
