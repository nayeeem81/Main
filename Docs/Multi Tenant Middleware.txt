using Microsoft.AspNetCore.Http;
using System.Linq;
using System.Threading.Tasks;

public class TenantResolverMiddleware
{
    private readonly RequestDelegate _next;

    public TenantResolverMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task InvokeAsync(
        HttpContext context, 
        ITenantRepository tenantRepository, 
        ITenantService tenantService)
    {
        // 1. Get the clean host name (e.g., "://mysapp.com" or "clientdomain.com")
        string host = context.Request.Host.Host.ToLower();

        if (!string.IsNullOrWhiteSpace(host) && host != "localhost")
        {
            Tenant? tenant = null;

            // 2. Identify the identifier (Is it a subdomain or a top-level custom domain?)
            string[] segments = host.Split('.');

            // Filter out 'www' if it's at the beginning
            if (segments.Length > 0 && segments[0] == "www")
            {
                segments = segments.Skip(1).ToArray();
                host = string.Join(".", segments);
            }

            // 3. Database Lookup Strategy
            if (segments.Length > 2)
            {
                // It's a subdomain lookup (e.g., "://myapp.com" -> looks up "tenant1")
                string subdomain = segments[0];
                tenant = await tenantRepository.GetBySubdomainAsync(subdomain);
            }
            
            // 4. Fallback to full top-level domain lookup if subdomain failed or wasn't present
            if (tenant == null)
            {
                // Looks up the full domain (e.g., "clientdomain.com")
                tenant = await tenantRepository.GetByCustomDomainAsync(host);
            }

            // 5. Store the found tenant into the scoped service
            if (tenant != null)
            {
                tenantService.CurrentTenant = tenant;
            }
        }

        // Continue the HTTP request pipeline
        await _next(context);
    }
}

using Microsoft.EntityFrameworkCore;
using System.Threading;
using System.Threading.Tasks;

public class AppDbContext : DbContext
{
    private readonly ITenantService _tenantService;
    private readonly string _currentTenantId;

    public AppDbContext(DbContextOptions<AppDbContext> options, ITenantService tenantService)
        : base(options)
    {
        _tenantService = tenantService;
        // Resolve the tenant ID safely. Fallback to empty if not resolved yet (e.g., migrations)
        _currentTenantId = _tenantService.CurrentTenant?.Id ?? string.Empty;
    }

    public DbSet<Product> Products => Set<Product>();

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // Configure all entities implementing IMustHaveTenant automatically
        foreach (var entityType in modelBuilder.Model.GetEntityTypes())
        {
            if (typeof(IMustHaveTenant).IsAssignableFrom(entityType.ClrType))
            {
                // Equivalent to: builder.Entity<Product>().HasQueryFilter(p => p.TenantId == _currentTenantId);
                modelBuilder.Entity(entityType.ClrType)
                    .HasQueryFilter(ConvertFilterExpression(entityType.ClrType));
            }
        }
    }

    // Automatically stamp incoming new records with the correct TenantId
    public override Task<int> SaveChangesAsync(CancellationToken cancellationToken = default)
    {
        foreach (var entry in ChangeTracker.Entries<IMustHaveTenant>())
        {
            if (entry.State == EntityState.Added)
            {
                entry.Entity.TenantId = _currentTenantId;
            }
        }
        return base.SaveChangesAsync(cancellationToken);
    }

    // Helper method to dynamically generate the lambda expression for EF Core
    private System.Linq.Expressions.LambdaExpression ConvertFilterExpression(System.Type type)
    {
        var parameter = System.Linq.Expressions.Expression.Parameter(type, "e");
        var property = System.Linq.Expressions.Expression.Property(parameter, nameof(IMustHaveTenant.TenantId));
        var comparison = System.Linq.Expressions.Expression.Equal(
            property, 
            System.Linq.Expressions.Expression.Property(System.Linq.Expressions.Expression.Constant(this), nameof(_currentTenantId))
        );
        return System.Linq.Expressions.Expression.Lambda(comparison, parameter);
    }
}

### How It Works in Practice

Queries: When you write _context.Products.ToList(), EF Core translates this to SQL behind the scenes as: 

SELECT * FROM Products WHERE TenantId = @tenantId.

Saves: When you run _context.Products.Add(new Product { Name = "Laptop" }), you do not need to provide a tenant ID. EF Core intercepts the save and automatically applies the logged-in tenant's ID before writing to the database.

Bypassing the filter: If a global admin ever needs to see data across all tenants, they can bypass the automation explicitly in their query by appending .IgnoreQueryFilters() (e.g., _context.Products.IgnoreQueryFilters().ToList()).

### Database per tenant andshared database, which is widely practised in read world and better perform?

Both architectures are widely used in the real world, but they solve completely different business and performance scaling challenges. Neither is universally "better"; instead, the choice depends heavily on your budget, data isolation requirements, and infrastructure strategy.

Here is a direct comparison of real-world practices and performance traits.

Shared Database (Multi-Tenant)All tenants share the same database tables and are separated by a TenantId column (using the EF Core query filters we built earlier).

Real-World Practice: This is the most common pattern for standard B2B SaaS apps (e.g., Slack, Trello, Shopify). It keeps operational overhead and infrastructure costs minimal when starting out or managing thousands of small clients.

Performance: 
High Performance for Density: You can fit thousands of small tenants onto a single, highly spec'd database instance, maximizing memory and CPU utilization.

The "Noisy Neighbor" Risk: If Tenant A runs a massive, unoptimized report, it consumes the shared database resources (CPU/RAM), causing latency and slow performance for Tenants B and C.

Scaling Limits: Once the shared tables reach hundreds of millions of rows, indexing, maintenance windows, and query performance degrade across the entire platform.

Database-Per-Tenant (Isolated)
Every tenant gets their own physical database instance or logical schema. The application swaps connection strings at runtime.

Real-World Practice: This is widely practiced in enterprise-grade software, healthcare, finance, or government systems where strict data privacy laws require absolute physical isolation, or for premium "Enterprise tier" clients who pay for dedicated resources.

Performance: 
Highest Predictable Performance: There is zero "Noisy Neighbor" effect. Tenant A's massive queries cannot slow down Tenant B because their database engines are completely separated.

Tailored Scaling: If one specific tenant grows massive, you can independently upgrade just their database to a higher tier without paying for a larger database for everyone else.

Connection Pool Overhead: The application server must maintain active database connection pools for hundreds of different databases simultaneously, which can heavily drain web server memory.

The Hybrid Real-World ApproachThe most mature, high-performing SaaS platforms use a Hybrid Approach:

Free-tier and standard users share a large, optimized Shared Database.

High-paying enterprise customers are given their own Dedicated Database via their custom domain, ensuring they receive maximum, isolated performance.
