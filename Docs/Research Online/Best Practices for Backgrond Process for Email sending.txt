Here is how to implement the Transactional Outbox Pattern for emails in your monolithic Clean Architecture app. 

This ensures your database updates and emails are processed reliably without slowing down user requests.

## 1. Core / Domain Layer: Define the Outbox Entity

Create a table structure to hold emails that are waiting to be sent.

namespace Domain.Entities;
public class EmailOutboxMessage
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public string TenantId { get; set; } = string.Empty; // Essential for multi-tenancy
    public string Receiver { get; set; } = string.Empty;
    public string Subject { get; set; } = string.Empty;
    public string Body { get; set; } = string.Empty;
    public DateTime CreatedOnUtc { get; set; } = DateTime.UtcNow;
    public DateTime? ProcessedOnUtc { get; set; }
    public string? Error { get; set; }
    public int RetryCount { get; set; }
}

## 2. Application Layer: Stage the Email inside a Transaction
When a business action happens (e.g., user registration), save both the user data and the outbox message inside the same database transaction.

namespace Application.Services;
public class UserService
{
    private readonly ApplicationDbContext _context;
    private readonly ITenantProvider _tenantProvider; // Injected to get current Tenant ID

    public UserService(ApplicationDbContext context, ITenantProvider tenantProvider)
    {
        _context = context;
        _tenantProvider = tenantProvider;
    }

    public async Task RegisterUserAsync(UserRegistrationDto dto)
    {
        // 1. Begin Database Transaction
        using var transaction = await _context.Database.BeginTransactionAsync();

        try
        {
            // 2. Save your core business entity
            var user = new User { Email = dto.Email, Name = dto.Name };
            _context.Users.Add(user);

            // 3. Stage the email in the Outbox (Do NOT send it yet)
            var outboxMessage = new EmailOutboxMessage
            {
                TenantId = _tenantProvider.GetCurrentTenantId(),
                Receiver = dto.Email,
                Subject = "Welcome!",
                Body = "<h1>Thank you for registering!</h1>" // Rendered from template
            };
            _context.EmailOutboxMessages.Add(outboxMessage);

            // 4. Commit together. If one fails, both roll back.
            await _context.SaveChangesAsync();
            await transaction.CommitAsync();
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }
}

## 3. Infrastructure Layer: Background Processor
Use a standard ASP.NET Core BackgroundService to poll the database, pick up unsent messages, and send them using tenant-specific credentials.

using Microsoft.Extensions.Hosting;using Microsoft.Extensions.DependencyInjection;
namespace Infrastructure.BackgroundJobs;
public class EmailOutboxProcessor : BackgroundService
{
    private readonly IServiceProvider _serviceProvider;

    public EmailOutboxProcessor(IServiceProvider serviceProvider)
    {
        _serviceProvider = serviceProvider;
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        while (!stoppingToken.IsCancellationRequested)
        {
            using (var scope = _serviceProvider.CreateScope())
            {
                var context = scope.ServiceProvider.GetRequiredService<ApplicationDbContext>();
                var emailSender = scope.ServiceProvider.GetRequiredService<ITenantEmailSender>();

                // Fetch top 20 unsent, non-failed messages
                var messages = await context.EmailOutboxMessages
                    .Where(m => m.ProcessedOnUtc == null && m.RetryCount < 3)
                    .OrderBy(m => m.CreatedOnUtc)
                    .Take(20)
                    .ToListAsync(stoppingToken);

                foreach (var message in messages)
                {
                    try
                    {
                        // Send using tenant credentials resolved inside ITenantEmailSender
                        await emailSender.SendAsync(
                            message.TenantId, 
                            message.Receiver, 
                            message.Subject, 
                            message.Body
                        );

                        message.ProcessedOnUtc = DateTime.UtcNow;
                    }
                    catch (Exception ex)
                    {
                        message.RetryCount++;
                        message.Error = ex.Message;
                    }
                }

                if (messages.Any())
                {
                    await context.SaveChangesAsync(stoppingToken);
                }
            }

            // Poll the database every 10 seconds
            await Task.Delay(TimeSpan.FromSeconds(10), stoppingToken);
        }
    }
}

## 4. Web Layer: Register the Worker
Register the background worker in your Program.cs.

builder.Services.AddHostedService<EmailOutboxProcessor>();
