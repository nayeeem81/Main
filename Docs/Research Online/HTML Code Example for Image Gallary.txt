Use code with caution.JavaScriptjavascriptdocument.addEventListener("DOMContentLoaded", () => {
  const mainImg = document.getElementById("main-product-img");
  const zoomContainer = document.getElementById("zoom-container");
  const thumbList = document.getElementById("thumb-list");
  const swatches = document.querySelectorAll(".swatch");
  const colorNameText = document.getElementById("color-name");

  // 1. Thumbnail Click Functionality
  thumbList.addEventListener("click", (e) => {
    if (e.target.classList.contains("thumb")) {
      document.querySelectorAll(".thumb").forEach(t => t.classList.remove("active"));
      e.target.classList.add("active");
      mainImg.src = e.target.dataset.full;
    }
  });

  // 2. Modern Desktop Hover-to-Zoom Effect
  zoomContainer.addEventListener("mousemove", (e) => {
    const rect = e.currentTarget.getBoundingClientRect();
    const x = ((e.clientX - rect.left) / rect.width) * 100;
    const y = ((e.clientY - rect.top) / rect.height) * 100;

    mainImg.style.transformOrigin = `${x}% ${y}%`;
    mainImg.style.transform = "scale(2)";
  });

  zoomContainer.addEventListener("mouseleave", () => {
    mainImg.style.transform = "scale(1)";
    mainImg.style.transformOrigin = "center center";
  });

  // 3. Dynamic Variant Swatches (Updates Images and Thumbnails)
  swatches.forEach(swatch => {
    swatch.addEventListener("click", (e) => {
      // Toggle Active Swatch
      swatches.forEach(s => s.classList.remove("active"));
      e.target.classList.add("active");

      // Update color label text
      colorNameText.textContent = e.target.dataset.color;

      // Parse variant images array
      const newImages = JSON.parse(e.target.dataset.images);

      // Set main product image to first item
      mainImg.src = newImages[0];

      // Rebuild thumbnails dynamically
      thumbList.innerHTML = newImages.map((imgUrl, index) => `
<img src="${imgUrl}"
     class="thumb ${index === 0 ? 'active' : ''}"
     data-full="${imgUrl}"
     alt="Product Variant ${index + 1}">
      `).join('');
    });
  });
});

Use code with caution.Why this approach is modern:Event Delegation: The thumbnail click logic uses a single event listener on the parent container (#thumb-list), optimizing memory usage.Inline Data Attributes: Product images for different colors are kept inside the dataset strings (data-images), mimicking how real backends deliver variant payload data.CSS Scale Transition: No secondary layout shift happens during zoom.
